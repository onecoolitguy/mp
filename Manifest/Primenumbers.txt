index.js
var app = {
    initialize: function() {
        document.addEventListener('deviceready', this.onDeviceReady.bind(this), false);
    },
    onDeviceReady: function() {
        console.log("Cordova is ready!");
    }
};
app.initialize();

function checkPrime() {
    var n = document.getElementById("num").value;

    if (n === "") {
        alert("Please enter a number");
        return;
    }

    n = parseInt(n);

    if (n <= 1) {
        document.getElementById("result").innerHTML = "<p>" + n + " is NOT a prime number</p>";
        return;
    }

    var isPrime = true;
    for (var i = 2; i <= Math.sqrt(n); i++) {
        if (n % i === 0) {
            isPrime = false;
            break;
        }
    }

    if (isPrime) {
        document.getElementById("result").innerHTML = "<p>" + n + " is a Prime number ✅</p>";
    } else {
        document.getElementById("result").innerHTML = "<p>" + n + " is NOT a Prime number ❌</p>";
    }
}
